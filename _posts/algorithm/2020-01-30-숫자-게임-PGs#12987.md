def solution(A, B):
    A.sort()
    B.sort()
    answer = 0
    aIdx = 0
    bIdx = 0
    
    while bIdx < len(B):
        if A[aIdx] < B[bIdx]:
            aIdx += 1
            bIdx += 1
            answer += 1
        else:
            bIdx += 1
    
    return answer
