#include <iostream>
#include <vector>

using namespace std;

int main() {

	int N, K;
	int cnt = 0;

	cin >> N >> K;

	vector<int> prime;
	prime.resize(N+1);

	for (int i = 2; i <= N; i++) {
		if (prime[i] == 1) continue;
		for (int j = i; j <= N; j += i) {
			if (prime[j] == 0) {
				cnt++;
				if (cnt == K) {
					cout << j;
					break;
				}
				prime[j] = 1;
			}
		}
	}

	return 0;
}
