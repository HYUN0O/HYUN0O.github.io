def move(now, _dir, visited):
    x = now[0]
    y = now[1]
    
    if _dir == "U":
        if now[1] != 10:
            now[1] += 1
            if visited[x][y][x][y+1] != 1 and visited[x][y+1][x][y] != 1:
                visited[x][y][x][y+1] = 1
                return 1
        
    elif _dir == "D":
        if now[1] != 0:
            now[1] -= 1
            if visited[x][y][x][y-1] != 1 and visited[x][y-1][x][y] != 1:
                visited[x][y][x][y-1] = 1
                return 1
        
    elif _dir == "R":
        if now[0] != 10:
            now[0] += 1
            if visited[x][y][x+1][y] != 1 and visited[x+1][y][x][y] != 1:
                visited[x][y][x+1][y] = 1
                return 1
        
    elif _dir == "L":
        if now[0] != 0:
            now[0] -= 1
            if visited[x][y][x-1][y] != 1 and visited[x-1][y][x][y] != 1:
                visited[x][y][x-1][y] = 1
                return 1
    return 0

def solution(dirs):
    answer = 0
    now = [5,5]
